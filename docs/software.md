interface SoftwareProduct {
  // === IDENTIFICATION ===
  id: string;
  category_id: string;           // e.g., "os/windows"
  
  // === BASIC INFO ===
  title: string;                 // "Microsoft 365 Personal - 1 Year"
  description: string;
  price: number;
  currency: 'BGN' | 'EUR' | 'USD';
  negotiable: boolean;
  
  // === PRODUCT IDENTIFICATION ===
  brand: string;                 // "Microsoft"
  product_name: string;          // "Microsoft 365"
  version?: string;              // "2024", "v22.0"
  edition?: string;              // "Personal", "Home", "Pro"
  
  // === LICENSE DETAILS ===
  license_type: LicenseType;
  license_duration: LicenseDuration;
  subscription_period?: string;  // "1 year", "1 month"
  users_included: number;        // 1, 5, unlimited
  devices_included: number;      // 1, 5, unlimited
  
  // === DELIVERY ===
  delivery_method: DeliveryMethod;
  instant_delivery: boolean;
  
  // === COMPATIBILITY ===
  platforms: Platform[];         // ["windows", "mac", "web"]
  os_requirements?: string;      // "Windows 10 or higher"
  system_requirements?: string;
  
  // === FEATURES ===
  features: string[];            // ["Cloud storage", "Mobile apps"]
  
  // === REGION ===
  region_lock: boolean;
  regions_available: string[];   // ["EU", "Global", "BG"]
  language: string[];            // ["en", "bg"]
  
  // === AUTHENTICITY ===
  is_genuine: boolean;
  seller_verified: boolean;
  
  // === SELLER INFO ===
  seller_type: 'private' | 'dealer' | 'authorized_reseller';
  location_city: string;
  
  // === LISTING META ===
  images: string[];
  featured: boolean;
  promoted: boolean;
  
  // === SYSTEM TAGS ===
  tags: string[];                // ["lifetime", "business", "subscription"]
}

// === ENUMS ===

type LicenseType = 
  | 'retail' | 'oem' | 'volume' | 'subscription' 
  | 'perpetual' | 'trial' | 'educational' | 'nfr';

type LicenseDuration = 
  | 'lifetime' | '1_year' | '2_years' | '3_years' 
  | '1_month' | '3_months' | '6_months';

type DeliveryMethod = 
  | 'digital_code' | 'email' | 'download_link' 
  | 'physical_box' | 'account_credentials';

type Platform = 
  | 'windows' | 'mac' | 'linux' | 'ios' | 'android' 
  | 'web' | 'chrome_os' | 'cross_platform';
