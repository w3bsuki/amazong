## Database Schema Additions

```sql
-- NOTE: Business profile fields already exist in private_profiles table
-- See: supabase/migrations/20260124120000_phase_c_private_surfaces.sql
-- DO NOT add to profiles table (security hardening)

-- Additional fields if needed (in private_profiles):
-- vat_number (already exists)
-- Add if missing:
ALTER TABLE private_profiles ADD COLUMN IF NOT EXISTS business_website VARCHAR(255);
ALTER TABLE private_profiles ADD COLUMN IF NOT EXISTS facebook_url VARCHAR(255);
ALTER TABLE private_profiles ADD COLUMN IF NOT EXISTS instagram_handle VARCHAR(100);
ALTER TABLE private_profiles ADD COLUMN IF NOT EXISTS physical_address TEXT;

-- Analytics: Daily aggregated views (NOT raw events - performance + privacy)
CREATE TABLE IF NOT EXISTS product_views_daily (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  product_id UUID REFERENCES products(id) ON DELETE CASCADE,
  seller_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  view_date DATE NOT NULL,
  source VARCHAR(50), -- 'search', 'direct', 'social', 'feed'
  view_count INTEGER DEFAULT 0,
  unique_viewers INTEGER DEFAULT 0, -- Anonymous count, no PII
  UNIQUE(product_id, view_date, source)
);

-- Seller-level daily aggregates (for dashboard overview)
CREATE TABLE IF NOT EXISTS seller_stats_daily (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  seller_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  stat_date DATE NOT NULL,
  total_views INTEGER DEFAULT 0,
  total_messages INTEGER DEFAULT 0,
  total_saves INTEGER DEFAULT 0,
  UNIQUE(seller_id, stat_date)
);

-- Export history
CREATE TABLE IF NOT EXISTS exports (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  profile_id UUID REFERENCES profiles(id),
  format VARCHAR(10), -- 'csv', 'json', 'xml'
  product_count INTEGER,
  file_url VARCHAR(500),
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

**Important:** No `viewer_id` stored (PII concern). Use anonymous counts only.
